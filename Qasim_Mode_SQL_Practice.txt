 SELECT year,
       month,
       west,
       south,
       midwest,
       northeast,
       west + south + midwest +northeast AS total
  FROM tutorial.us_housing_units
  SELECT year,
       month,
       west,
       south,
       midwest,
       northeast
  FROM tutorial.us_housing_units
  WHERE west > (midwest + northeast)
  SELECT year,
       month,
       west/(west + south + midwest + northeast)*100 AS west_prcnt,
       south/(west + south + midwest + northeast)*100 AS south_prcnt,
       midwest/(west + south + midwest + northeast)*100 AS midwest_prcnt,
       northeast/(west + south + midwest + northeast)*100 AS northeast_prcnt
  FROM tutorial.us_housing_units
 WHERE year >= 2000
